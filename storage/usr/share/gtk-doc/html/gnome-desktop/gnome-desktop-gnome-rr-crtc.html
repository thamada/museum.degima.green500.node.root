typedef struct GnomeRRCrtc GnomeRRCrtc;

enum GnomeRRRotation

typedef enum
{
    GNOME_RR_ROTATION_0 = (1 << 0),
    GNOME_RR_ROTATION_90 = (1 << 1),
    GNOME_RR_ROTATION_180 = (1 << 2),
    GNOME_RR_ROTATION_270 = (1 << 3),
    GNOME_RR_REFLECT_X = (1 << 4),
    GNOME_RR_REFLECT_Y = (1 << 5)
} GnomeRRRotation;
